import Button from "@space-kit/button"
import { graphql } from "gatsby"

export const query = graphql`
  query {
    allComponentMetadata(filter: { displayName: { eq: "Button" } }) {
      edges {
        node {
          id
          displayName
          props {
            id
            name
            tsType
          }
        }
      }
    }
  }
`

## Button

---

<Button>Hello Button</Button>

### Installation

Install button component from npm

```bash
npm install --save @space-kit/button
```

### Usage

Add button component to your file

```jsx
import React from "react"
import Button from "@space-kit/button"

const MyComponent = () => {
  return <Button>Hello Button</Button>
}
```
