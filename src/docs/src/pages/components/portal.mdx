import Portal from "@space-kit/portal"
import { graphql } from "gatsby"

export const query = graphql`
  query {
    allComponentMetadata(filter: { displayName: { eq: "Portal" } }) {
      edges {
        node {
          id
          displayName
          props {
            id
            name
            tsType
          }
        }
      }
    }
  }
`

## Portal

---

### Installation

Install portal component from npm

```bash
npm install --save @space-kit/portal
```

### Usage

Add portal component to your file

```jsx
import React from "react"
import Portal from "@space-kit/portal"

const MyComponent = () => {
  return <Porta>Hello Portal</Porta>
}
```
