import { AlignLeftIcon } from '@space-kit/icons';
import { graphql } from "gatsby"

export const query = graphql`
  query {
    allComponentMetadata(filter: { displayName: { eq: "Icon" } }) {
      edges {
        node {
          id
          displayName
          props {
            id
            name
            tsType
          }
        }
      }
    }
  }
`

## Icons

---

<AlignLeftIcon size={16} />

### Installation

Install icons component from npm

```bash
npm install --save @space-kit/icons
```

### Usage

Add icons component to your file

```jsx
import React from "react"
import { AlignLeftIcon } from '@space-kit/icons';

const MyComponent = () => {
  return <AlignLeftIcon size={16} />
}
```
